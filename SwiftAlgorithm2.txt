func spiralTraverse(_ matrix:[[Int]])->[Int] {
    guard !matrix.isEmpty else {
        return []
    }
    var result:[Int] = [Int]()
    let total = matrix[0].count * matrix.count
    
    var lefter = 0
    var top = 0
    var righter = matrix[0].count - 1
    var bottom = matrix.count - 1
    
    while (result.count < total) {
        
        //go left to right
        for col in stride(from: lefter,through: righter, by:1){
            result.append(matrix[top][col])
        }
        //go top bottom
        for row in stride(from: top+1,through: bottom, by:1){
            result.append(matrix[row][righter])
        }
        //go right to left
        
        //go bottom top
        
        lefter += 1
        top += 1
        righter -= 1
        bottom -= 1
    }
    
    return result
}
